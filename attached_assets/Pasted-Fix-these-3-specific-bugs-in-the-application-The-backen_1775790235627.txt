export const getDoctorRankLabel = (rank: string | null | undefined, isArabic: boolean = false): string => {
  if (!rank) return 'N/A';
  const ranks: Record<string, { en: string; ar: string }> = {
    'GP': { en: 'General Practitioner', ar: 'طبيب عام' },
    'RESIDENT': { en: 'Resident', ar: 'طبيب مقيم' },
    'SPECIALIST': { en: 'Specialist', ar: 'اختصاصي' },
    'CONSULTANT': { en: 'Consultant', ar: 'استشاري' },
  };
  return ranks[rank]?.[isArabic ? 'ar' : 'en'] ?? rank;
};
